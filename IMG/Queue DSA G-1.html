#include <iostream>
#include <queue>
#include <string>

using namespace std;

// Structure for Print Job
struct PrintJob {
    int jobID;
    string fileName;
    int pages;
};

// Queue for print jobs
queue<PrintJob> printQueue;

// Function to add a print job
void addJob() {
    PrintJob job;

    cout << "\nEnter Job ID: ";
    cin >> job.jobID;

    cin.ignore();
    cout << "Enter File Name: ";
    getline(cin, job.fileName);

    cout << "Enter Number of Pages: ";
    cin >> job.pages;

    printQueue.push(job);

    cout << "Print job added successfully!\n";
}

// Function to process (print) a job
void processJob() {
    if (printQueue.empty()) {
        cout << "\nNo print jobs in queue.\n";
        return;
    }

    PrintJob current = printQueue.front();
    printQueue.pop();

    cout << "\nProcessing Print Job...\n";
    cout << "Job ID   : " << current.jobID << endl;
    cout << "File Name: " << current.fileName << endl;
    cout << "Pages    : " << current.pages << endl;
    cout << "Print completed successfully!\n";
}

// Function to cancel a job
void cancelJob() {
    if (printQueue.empty()) {
        cout << "\nQueue is empty.\n";
        return;
    }

    int id;
    cout << "\nEnter Job ID to cancel: ";
    cin >> id;

    queue<PrintJob> tempQueue;
    bool found = false;

    while (!printQueue.empty()) {
        PrintJob current = printQueue.front();
        printQueue.pop();

        if (current.jobID == id) {
            found = true;
            cout << "Print job cancelled successfully!\n";
        } else {
            tempQueue.push(current);
        }
    }

    printQueue = tempQueue;

    if (!found) {
        cout << "Job ID not found.\n";
    }
}

// Function to display all jobs
void displayJobs() {
    if (printQueue.empty()) {
        cout << "\nNo print jobs in queue.\n";
        return;
    }

    queue<PrintJob> tempQueue = printQueue;

    cout << "\nCurrent Print Queue:\n";
    cout << "-------------------------\n";

    while (!tempQueue.empty()) {
        PrintJob current = tempQueue.front();

        cout << "Job ID   : " << current.jobID << endl;
        cout << "File Name: " << current.fileName << endl;
        cout << "Pages    : " << current.pages << endl;
        cout << "-------------------------\n";

        tempQueue.pop();
    }
}

// Main function
int main() {
    int choice;

    do {
        cout << "\n===== PRINT QUEUE MANAGEMENT SYSTEM =====\n";
        cout << "1. Add Print Job\n";
        cout << "2. Process Print Job\n";
        cout << "3. Cancel Print Job\n";
        cout << "4. Display Print Queue\n";
        cout << "5. Exit\n";
        cout << "Enter your choice: ";
        cin >> choice;

        switch (choice) {
            case 1:
                addJob();
                break;

            case 2:
                processJob();
                break;

            case 3:
                cancelJob();
                break;

            case 4:
                displayJobs();
                break;

            case 5:
                cout << "\nExiting Program...\n";
                break;

            default:
                cout << "\nInvalid choice. Try again.\n";
        }

    } while (choice != 5);

    return 0;
}
